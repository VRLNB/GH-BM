-- 注入器文件操作兼容（核心：获取持久化路径）
local readFunc, writeFunc, getPathFunc
pcall(function()
    if type(syn) == "table" then
        readFunc = syn.readfile or readfile
        writeFunc = syn.writefile or writefile
        getPathFunc = syn.datapath
    else
        readFunc = readfile or readFile
        writeFunc = writefile or writeFile
        getPathFunc = function()
            return "Roblox/Scripts"
        end
    end
end)

-- ================ 全局终止标记（修复粒子循环泄漏） ================
local isParticleRunning = false

-- ================ 验证文件夹核心逻辑 ================
local function getVerifyFolderPath()
    local basePath
    if getPathFunc then
        basePath = getPathFunc()
    else
        basePath = "C:/RobloxScripts"
        pcall(function()
            if game:GetService("RunService"):IsStudio() then
                basePath = "Roblox/Scripts"
            elseif string.find(MarketplaceService:GetProductInfo(1).Name, "Mac") then
                basePath = "~/Library/Roblox/Scripts"
            end
        end)
    end
    return basePath .. "/同意MIMI脚本"
end

local function folderExists()
    if not readFunc then
        return false
    end
    local folderPath = getVerifyFolderPath()
    local success = pcall(function()
        readFunc(folderPath .. "/.verify")
    end)
    return success
end

local function createTargetFolder()
    if not writeFunc then
        return
    end
    local folderPath = getVerifyFolderPath()
    local success = pcall(function()
        writeFunc(folderPath .. "/.verify", "")
        print("验证文件夹创建成功：" .. folderPath)
    end)
    if not success then
        local tempFolder = Instance.new("Folder")
        tempFolder.Name = "同意MIMI脚本"
        tempFolder.Parent = PlayerGui
        print("警告：持久化文件夹创建失败，使用临时文件夹（仅当前会话有效）")
    end
end

local function createFrame(parent, size, position)
    local frame = Instance.new("Frame")
    frame.Size = size
    frame.Position = position
    frame.BackgroundColor3 = Color3.fromRGB(60, 60, 80)
    frame.BackgroundTransparency = 0.3
    frame.BorderSizePixel = 1
    frame.BorderColor3 = Color3.fromRGB(220, 240, 255)
    frame.Parent = parent
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, 12)
    corner.Parent = frame
    return frame
end

local function createClickToInput(parent, size, position, triggerText, placeholderText)
    local trigger = Instance.new("TextButton")
    trigger.Size = size
    trigger.Position = position
    trigger.BackgroundColor3 = Color3.fromRGB(80, 80, 100)
    trigger.BackgroundTransparency = 0.7
    trigger.Text = triggerText
    trigger.TextColor3 = Color3.new(1, 1, 1)
    trigger.TextSize = 14
    trigger.Parent = parent
    local cornerTrigger = Instance.new("UICorner")
    cornerTrigger.CornerRadius = UDim.new(0, 6)
    cornerTrigger.Parent = trigger
    local box = Instance.new("TextBox")
    box.Size = size
    box.Position = position
    box.BackgroundColor3 = Color3.fromRGB(80, 80, 100)
    box.BackgroundTransparency = 0.7
    box.TextColor3 = Color3.new(1, 1, 1)
    box.TextSize = 14
    box.PlaceholderText = placeholderText
    box.Visible = false
    box.Parent = parent
    local cornerBox = Instance.new("UICorner")
    cornerBox.CornerRadius = UDim.new(0, 6)
    cornerBox.Parent = box
    trigger.MouseButton1Click:Connect(function()
        trigger.Visible = false
        box.Visible = true
        box:CaptureFocus()
    end)
    return trigger, box
end

local function playTempSound(soundId)
    local fullSoundId = "rbxassetid://" .. tostring(soundId)
    local sound = Instance.new("Sound")
    sound.SoundId = fullSoundId
    sound.Volume = 1.0
    sound.Parent = workspace
    pcall(function()
        sound:Play()
        sound.Ended:Connect(function()
            sound:Destroy()
        end)
        task.delay(5, function()
            if sound.Parent then
                sound:Destroy()
            end
        end)
    end)
end

-- ================ 悬浮窗+背景粒子特效 ================
local floatWindow
local function createParticleFloatingWindow()
    local floatGui = Instance.new("ScreenGui")
    floatGui.Name = "ParticleFloatWindow"
    floatGui.IgnoreGuiInset = true
    floatGui.ZIndexBehavior = Enum.ZIndexBehavior.Global
    floatGui.Parent = PlayerGui
    floatWindow = floatGui

    local floatFrame = createFrame(floatGui, UDim2.new(0, 280, 0, 160), UDim2.new(0.05, 0, 0.1, 0))
    floatFrame.BackgroundColor3 = Color3.fromRGB(40, 40, 60)
    floatFrame.BackgroundTransparency = 0.2
    floatFrame.BorderColor3 = Color3.fromRGB(255, 210, 0)

    createLabel(floatFrame, "MIMI脚本悬浮窗", UDim2.new(1, 0, 0, 30), UDim2.new(0, 0, 0, 10))
    createLabel(floatFrame, "验证通过后自动加载脚本", UDim2.new(1, 0, 0, 20), UDim2.new(0, 0, 0, 45))

    -- 拖动功能
    local isDragging = false
    local dragStartPos, windowStartPos
    floatFrame.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            isDragging = true
            dragStartPos = input.Position
            windowStartPos = floatFrame.Position
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if isDragging and input.UserInputType == Enum.UserInputType.MouseMovement then
            local delta = input.Position - dragStartPos
            floatFrame.Position = UDim2.new(
                windowStartPos.X.Scale, windowStartPos.X.Offset + delta.X,
                windowStartPos.Y.Scale, windowStartPos.Y.Offset + delta.Y
            )
        end
    end)

    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            isDragging = false
        end
    end)

    -- 粒子生成（添加终止标记）
    local function spawnBackgroundParticles()
        isParticleRunning = true -- 启动标记
        while isParticleRunning and floatGui.Parent do
            local particle = Instance.new("Frame")
            particle.Size = UDim2.new(0, math.random(4, 8), 0, math.random(4, 8))
            particle.Position = UDim2.new(
                math.random(0, 100)/100, -particle.AbsoluteSize.X/2,
                math.random(0, 100)/100, -particle.AbsoluteSize.Y/2
            )
            particle.AnchorPoint = Vector2.new(0.5, 0.5)
            particle.BackgroundColor3 = Color3.fromRGB(
                math.random(220, 255), math.random(180, 220), math.random(0, 50)
            )
            particle.BackgroundTransparency = 0.3
            particle.Parent = floatFrame
            local corner = Instance.new("UICorner", particle)
            corner.CornerRadius = UDim.new(1, 0)

            local targetPos = UDim2.new(
                math.random(0, 100)/100, -particle.AbsoluteSize.X/2,
                math.random(0, 100)/100, -particle.AbsoluteSize.Y/2
            )
            local tween = TweenService:Create(
                particle,
                TweenInfo.new(math.random(8, 15), Enum.EasingStyle.Linear),
                {Position = targetPos}
            )
            tween:Play()

            task.delay(math.random(10, 18), function()
                if particle.Parent then
                    TweenService:Create(
                        particle,
                        TweenInfo.new(2),
                        {BackgroundTransparency = 1}
                    ):Play()
                    task.wait(2)
                    particle:Destroy()
                end
            end)
            task.wait(math.random(0.3, 0.8))
        end
    end

    coroutine.wrap(spawnBackgroundParticles)()

    -- F5显示/隐藏
    local isVisible = true
    UserInputService.InputBegan:Connect(function(input)
        if input.KeyCode == Enum.KeyCode.F5 then
            isVisible = not isVisible
            floatFrame.Visible = isVisible
        end
    end)
end

-- ================ 验证完成判定+悬浮窗销毁 ================
local function onAllVerificationCompleted()
    print("【MIMI验证】所有验证已完成，开始销毁悬浮窗...")
    isParticleRunning = false -- 终止粒子循环（核心修复）

    if not floatWindow or not floatWindow.Parent then
        print("【MIMI验证】悬浮窗已不存在，无需销毁")
        return
    end

    local floatFrame = floatWindow:FindFirstChildOfClass("Frame")
    if floatFrame then
        TweenService:Create(
            floatFrame,
            TweenInfo.new(1.2, Enum.EasingStyle.Quad),
            {BackgroundTransparency = 1, BorderTransparency = 1}
        ):Play()
    end

    for _, child in ipairs(floatWindow:GetDescendants()) do
        if child:IsA("Frame") and child:FindFirstChildOfClass("UICorner") then
            TweenService:Create(
                child,
                TweenInfo.new(0.8, Enum.EasingStyle.Quad),
                {BackgroundTransparency = 1}
            ):Play()
            task.wait(0.1)
        end
    end

    task.delay(1.5, function()
        if floatWindow and floatWindow.Parent then
            floatWindow:Destroy()
            print("【MIMI验证】悬浮窗已完全销毁")
        end
    end)
end

local function Popup_VerifyRelated()

    createLabel(frame, "谁最帅😎", UDim2.new(1, 0, 0, 35), UDim2.new(0, 0, 0, 25))

        frame, UDim2.new(1, -50, 0, 45), UDim2.new(0, 25, 0, 75),
        "点击输入答案", "请输入正确答案"
    )
    local statusLabel = createLabel(frame, "", UDim2.new(1, 0, 0, 25), UDim2.new(0, 0, 0, 135))
    local done = false

    createButton(frame, "验证答案", UDim2.new(0.45, 0, 0, 35), UDim2.new(0.1, 0, 0, 175), function()
        if inputRelated.Text == "MIMI专属答案" then
            done = true
            gui:Destroy()
        else
            statusLabel.Text = "答案错误，无法使用MIMI脚本哦😡"
            statusLabel.TextColor3 = Color3.fromRGB(255, 60, 60)
        end
    end)

    createButton(frame, "退出脚本", UDim2.new(0.45, 0, 0, 35), UDim2.new(0.5, 0, 0, 175), function()
        done = true
        gui:Destroy()
        error("用户主动退出，脚本停止运行")
    end)

    -- 超时保护（5分钟）
    local timeout = 300
    local start = os.time()
    while not done and os.time() - start < timeout do
        task.wait()
    end
    if not done then
        gui:Destroy()
        error("验证超时，脚本终止")
    end
end

    local httpSuccess, httpResult = pcall(function()

            loadstring(httpResult)()

            task.wait(6)
            errGui:Destroy()
            warn("【MIMI验证】执行失败：", execErr)
        end
    else

        local errFrame = createFrame(errGui, UDim2.new(0, 400, 0, 120), UDim2.new(0.5, -200, 0.5, -60))

-- ================ 右下角弹窗逻辑 ======================
local NOTIFICATION_DATA = {
    maxCount = 5,
    spacing = 6,
    width = 120,
    height = 60,
    activeWindows = {}
}

local function updateWindowPositions()
    for index, windowData in ipairs(NOTIFICATION_DATA.activeWindows) do
        local frame = windowData.frame
        if frame and frame:IsDescendantOf(game) then
            local targetY = -NOTIFICATION_DATA.height - 10 - (NOTIFICATION_DATA.height + NOTIFICATION_DATA.spacing) * (index - 1)
            frame.Position = UDim2.new(
                1, -NOTIFICATION_DATA.width - 2,
                1, targetY
            )
        end
    end
end

local function showNotification(title, text, duration)
    local localPlayer = Players.LocalPlayer
    if not localPlayer then
        return
    end
    local playerGui = localPlayer:FindFirstChild("PlayerGui")
    if not playerGui then
        return
    end
    if #NOTIFICATION_DATA.activeWindows >= NOTIFICATION_DATA.maxCount then
        local oldestWindow = NOTIFICATION_DATA.activeWindows[#NOTIFICATION_DATA.activeWindows]
        if oldestWindow and oldestWindow.gui:IsDescendantOf(game) then
            oldestWindow.gui:Destroy()
        end
        table.remove(NOTIFICATION_DATA.activeWindows, #NOTIFICATION_DATA.activeWindows)
    end
    for i = #NOTIFICATION_DATA.activeWindows, 1, -1 do
        NOTIFICATION_DATA.activeWindows[i].index = i + 1
    end
    local gui = createScreenGui("Notification_" .. #NOTIFICATION_DATA.activeWindows + 1)
    local frame = createFrame(gui, UDim2.new(0, NOTIFICATION_DATA.width, 0, NOTIFICATION_DATA.height), UDim2.new(1, -NOTIFICATION_DATA.width - 2, 1, -NOTIFICATION_DATA.height - 10))
    createLabel(frame, title, UDim2.new(1, 0, 0, 20), UDim2.new(0, 5, 0, 5))
    createLabel(frame, text, UDim2.new(1, 0, 0, 30), UDim2.new(0, 5, 0, 25))
    table.insert(NOTIFICATION_DATA.activeWindows, 1, {
        gui = gui,
        frame = frame,
        index = 1
    })
    task.delay(duration, function()
        for i, windowData in ipairs(NOTIFICATION_DATA.activeWindows) do
            if windowData.gui == gui then
                windowData.gui:Destroy()
                table.remove(NOTIFICATION_DATA.activeWindows, i)
                updateWindowPositions()
                break
            end
        end
    end)
    updateWindowPositions()
end
